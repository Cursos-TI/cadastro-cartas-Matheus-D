
#include <stdio.h>

/*
  Super Trunfo - Países (Nível Aventureiro)
  - Lê dados de DUAS cartas (sem for/while, sem if/else)
  - Calcula e exibe Densidade Populacional e PIB per Capita
  - PIB de entrada em BILHÕES de reais; per capita em reais/habitante
  - Formatação com duas casas decimais
*/

int main() {
    /* --------- Carta 1 --------- */
    char estado1;                // letra de 'A' a 'H'
    char codigo1[4];             // ex.: "A01" (3 chars + '\0')
    char cidade1[100];           // nome da cidade (string)
    int populacao1;              // habitantes
    float area1;                 // km²
    float pib1;                  // PIB total em BILHÕES de reais
    int pontos1;                 // número de pontos turísticos
    float densidade1;            // hab/km² (calculado)
    float pib_per_capita1;       // reais/hab (calculado)

    /* --------- Carta 2 --------- */
    char estado2;
    char codigo2[4];
    char cidade2[100];
    int populacao2;
    float area2;
    float pib2;
    int pontos2;
    float densidade2;
    float pib_per_capita2;

    /* ========== Entrada de dados (Carta 1) ========== */
    printf("=== Cadastro da Carta 1 ===\n");

    printf("Estado (A-H): ");
    scanf(" %c", &estado1);

    printf("Codigo da Carta (ex.: A01): ");
    scanf("%3s", codigo1);

    printf("Nome da Cidade: ");
    scanf(" %99[^\n]", cidade1);

    printf("Populacao (habitantes): ");
    scanf("%d", &populacao1);

    printf("Area (km2): ");
    scanf("%f", &area1);

    printf("PIB (em BILHOES de reais): ");
    scanf("%f", &pib1);

    printf("Numero de Pontos Turisticos: ");
    scanf("%d", &pontos1);

    /* ========== Cálculos (Carta 1) ========== */
    densidade1 = (float)populacao1 / area1;                 // hab/km²
    pib_per_capita1 = (pib1 * 1e9f) / (float)populacao1;    // reais/hab

    /* ========== Entrada de dados (Carta 2) ========== */
    printf("\n=== Cadastro da Carta 2 ===\n");

    printf("Estado (A-H): ");
    scanf(" %c", &estado2);

    printf("Codigo da Carta (ex.: B02): ");
    scanf("%3s", codigo2);

    printf("Nome da Cidade: ");
    scanf(" %99[^\n]", cidade2);

    printf("Populacao (habitantes): ");
    scanf("%d", &populacao2);

    printf("Area (km2): ");
    scanf("%f", &area2);

    printf("PIB (em BILHOES de reais): ");
    scanf("%f", &pib2);

    printf("Numero de Pontos Turisticos: ");
    scanf("%d", &pontos2);

    /* ========== Cálculos (Carta 2) ========== */
    densidade2 = (float)populacao2 / area2;
    pib_per_capita2 = (pib2 * 1e9f) / (float)populacao2;

    /* ========== Saída formatada ========== */
    printf("\n\nCarta 1:\n");
    printf("Estado: %c\n", estado1);
    printf("Codigo: %s\n", codigo1);
    printf("Nome da Cidade: %s\n", cidade1);
    printf("Populacao: %d\n", populacao1);
    printf("Area: %.2f km²\n", area1);
    printf("PIB: %.2f bilhoes de reais\n", pib1);
    printf("Numero de Pontos Turisticos: %d\n", pontos1);
    printf("Densidade Populacional: %.2f hab/km²\n", densidade1);
    printf("PIB per Capita: %.2f reais\n", pib_per_capita1);

    printf("\nCarta 2:\n");
    printf("Estado: %c\n", estado2);
    printf("Codigo: %s\n", codigo2);
    printf("Nome da Cidade: %s\n", cidade2);
    printf("Populacao: %d\n", populacao2);
    printf("Area: %.2f km²\n", area2);
    printf("PIB: %.2f bilhoes de reais\n", pib2);
    printf("Numero de Pontos Turisticos: %d\n", pontos2);
    printf("Densidade Populacional: %.2f hab/km²\n", densidade2);
    printf("PIB per Capita: %.2f reais\n", pib_per_capita2);

    return 0;
}
